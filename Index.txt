/* ===============================
   LOGIQUE DU CODEUR PRO
   =============================== */

let motDePasse = "abc";
let motDePasseChange = false;

/* ====== SUBSTITUTION ====== */
function genererSubstitution(cle) {
    const alphabet = "abcdefghijklmnopqrstuvwxyz";
    let seed = 0;

    for (let c of cle) seed += c.charCodeAt(0);

    let arr = alphabet.split("");
    for (let i = 0; i < arr.length; i++) {
        let j = (i * 7 + seed) % arr.length;
        [arr[i], arr[j]] = [arr[j], arr[i]];
    }

    let map = {};
    for (let i = 0; i < alphabet.length; i++) {
        map[alphabet[i]] = arr[i];
    }
    return map;
}

/* ====== ENCODAGE ====== */
function encoder(message, cle = motDePasse) {
    const map = genererSubstitution(cle);
    const alphabet = "abcdefghijklmnopqrstuvwxyz";
    let sortie = [];

    message = message.toLowerCase();

    for (let i = 0; i < message.length; i++) {
        let c = message[i];
        if (!alphabet.includes(c)) continue;

        let sub = map[c];
        let r = Math.floor(Math.random() * 10);

        let bloc = (i % 2 === 0) ? (r + sub) : (sub + r);
        sortie.push(bloc);

        if ((i + 1) % 3 === 0) {
            let bruit = String.fromCharCode(97 + Math.random() * 26);
            bruit += Math.floor(Math.random() * 10);
            bruit += String.fromCharCode(97 + Math.random() * 26);
            sortie.push(bruit);
        }
    }

    return sortie.join(" ");
}

/* ====== D√âCODAGE ====== */
function decoder(code, cle = motDePasse) {
    const map = genererSubstitution(cle);
    const reverseMap = {};
    for (let k in map) reverseMap[map[k]] = k;

    const blocs = code.split(" ").filter(b => b.trim() !== "");
    let resultat = "";
    let vraiBlocIndex = 0;

    for (let bloc of blocs) {
        if ((vraiBlocIndex + 1) % 4 === 0) {
            vraiBlocIndex++;
            continue;
        }

        let lettre = bloc.replace(/[0-9]/g, "");
        if (reverseMap[lettre]) resultat += reverseMap[lettre];

        vraiBlocIndex++;
    }

    return resultat;
}

/* ====== CHANGEMENT DE MOT DE PASSE ====== */
function changerMotDePasse(ancien, nouveau) {
    if (!nouveau) return "Le nouveau mot de passe est vide.";

    if (!motDePasseChange) {
        motDePasse = nouveau;
        motDePasseChange = true;
        return "Mot de passe d√©fini pour la premi√®re fois.";
    }

    if (ancien !== motDePasse) return "Ancien mot de passe incorrect.";

    motDePasse = nouveau;
    return "Mot de passe chang√©.";
}

    const mp = localStorage.getItem("motDePassePro");
    const mpChange = localStorage.getItem("motDePasseChange");

    if (mp) motDePasse = mp;
    if (mpChange === "true") motDePasseChange = true;

    document.getElementById("motActuel").value = motDePasse;
});

function validerAcces() {
    const entree = document.getElementById("motPasseEntree").value;

    if (entree === motDePasse) {
        document.getElementById("zoneAcces").classList.add("hidden");
        document.getElementById("zoneCodeur").classList.remove("hidden");
    } else {
        document.getElementById("statusAcces").textContent = "Mot de passe incorrect.";
    }
}

function uiEncoder() {
    const clair = document.getElementById("messageClair").value;
    document.getElementById("messageCode").value = encoder(clair);
}

function uiDecoder() {
    const code = document.getElementById("messageCode").value;
    document.getElementById("messageClair").value = decoder(code);
}

function uiChangerMotDePasse() {
    const ancien = document.getElementById("ancienMot").value;
    const nouveau = document.getElementById("nouveauMot").value;

    const msg = changerMotDePasse(ancien, nouveau);
    document.getElementById("statusMot").textContent = msg;
    document.getElementById("motActuel").value = motDePasse;

    localStorage.setItem("motDePassePro", motDePasse);
    localStorage.setItem("motDePasseChange", motDePasseChange);

    document.getElementById("ancienMot").value = "";
    document.getElementById("nouveauMot").value = "";
}
